+const MAX_RETRIES = 5;
+const BASE_DELAY_MS = 1000;
+
+interface RetryEntry {
+  url: string;
+  payload: object;
+  attempt: number;
+  nextRetryAt: number;
+}
+
+const retryQueue: RetryEntry[] = [];
+
+export async function scheduleRetry(
+  url: string,
+  payload: object,
+  statusCode: number,
+): Promise<void> {
+  const attempt = 1;
+  const delay = BASE_DELAY_MS * Math.pow(2, attempt - 1);
+  retryQueue.push({
+    url,
+    payload,
+    attempt,
+    nextRetryAt: Date.now() + delay,
+  });
+  console.log(
+    `Scheduled retry #${attempt} for ${url} (status ${statusCode}) in ${delay}ms`,
+  );
+}
+
+export async function processRetryQueue(): Promise<void> {
+  const now = Date.now();
+  const due = retryQueue.filter((e) => e.nextRetryAt <= now);
+  for (const entry of due) {
+    // re-attempt the webhook send
+    const idx = retryQueue.indexOf(entry);
+    retryQueue.splice(idx, 1);
+  }
+}
